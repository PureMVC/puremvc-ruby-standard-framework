  def remove_mediator(mediator_name)
    mediator = @mediator_map[mediator_name]
    if mediator
      mediator.list_notification_interests.each do |interest|
        remove_observer(interest, mediator)
      end
      @mediator_map.delete(mediator_name)
      mediator.on_remove
    end
    mediator
  end
